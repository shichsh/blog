JS Prototype

Each object has a prototype and inherits all the properties from its prototype.
The object prototype is also an object and thus has its own prototype. Thus we get a prototype chain until Object.prototype, whose prototype is null.

var assert = require("assert");

/*
Object and String are functions. String is not a subclass of Object. There is no class definition is JS. 
Inheritance is between objects
top level object: Object.prototype
*/

// Why do we need prototype
function A() {
    this.value = 10;
    this.getValue = function() {
        return this.value;
    }
}

var a1 = new A();
var a2 = new A();
assert.equal(a1.value, a2.value);
assert.ok(!(a1.getValue==a2.getValue));  // getValue is wasteful if creates many objects

function B() {
    this.value = 20;
}
B.prototype.getValue = function() {
    return this.value;
}
var b1 = new B();
var b2 = new B();
assert.equal(b1.value, b2.value);
assert.ok(b1.getValue==b2.getValue);  // share one copy, save space capacity
