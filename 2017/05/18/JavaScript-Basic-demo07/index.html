JS Prototype (continued)

var assert = require("assert");

//Internal process of Object.create
function create(obj) {
    var F = function() {}
    F.prototype = obj;
    return new F();
}

function Base() {}
Base.prototype.value = 10;
function Sub() {}

Sub.prototype = create(Base.prototype);
Sub.prototype.constructor = Sub;

var sub = new Sub();
// Prototype chain: sub --> Sub.prototype (=new F() --> F.prototype =) --> 
// Base.prototype --> Object.prototype --> null
assert.ok(sub instanceof Base);
assert.equal(10, sub.value);

// Inheritance between objects
var a = {
    x: 100
};
var b = {};
b.__proto__ = a;      // Make b inherits a
// Prototype chain: b --> a --> Object.prototype --> null
assert.equal(100, b.x);

assert.ok(sub.__proto__===Sub.prototype);
assert.ok(Sub.__proto__===Function.prototype);
assert.ok(Sub.prototype.isPrototypeOf(sub));
assert.ok(Function.prototype.isPrototypeOf(Sub));
assert.ok(a.isPrototypeOf(b));

// An example
function Shape() {
    this.x = 0;
    this.y = 0;
}
Shape.prototype.move = function(x, y) {
    this.x = x;
    this.y = y;
}

function Triangle() {
    Shape.call(this);
    console.log("Create an instance of Triangle");
}
Triangle.prototype = Object.create(Shape.prototype);
Triangle.prototype.constructor = Triangle;

var t = new Triangle();
// Prototype chain: t --> Triangle.prototype --> Shape.prototype --> Object.prototype
assert.equal(0, t.x);
assert.equal(0, t.y);

function Rectangle() {
    this.base = Shape; // whatever base is. should have "this.base" to call Shape, not Shape()
    this.base();
    console.log("Create an instance of Rectangle");
}
Rectangle.prototype = Object.create(Shape.prototype);
Rectangle.prototype.constructor = Rectangle;

var r = new Rectangle();
assert.equal(0, r.x);
assert.equal(0, r.y);

function Point() {}
Point.prototype.move = function(x, y) {
    Shape.prototype.move.call(this, x, y);
}

var p = new Point();
p.move(10, 20);
console.log(Object.keys(p));
assert.equal(10, p.x);
assert.equal(20, p.y);

Dependency Injection

function foo() {
    this.add = function() {
        return this.x + this.y;
    }
}

function goo() {
    this.x = 10;
    this.y = 20;
}

var obj = {};
goo.call(obj);
foo.call(obj);
assert.equal(30, obj.add());
// if function call(obj), then obj replace "this" in function
// "this" is not constrict to the constructor.
