JS Property Attributes and Object Status

A property has 3 attributes enumerable, writable, configurable

```JavaScript
var assert = require(“assert”);

var alice = {
name: “Alice”
};
alice.age = 30; // By default, age is enumerable, writable and configurable

console.log(Object.keys(alice)); // age is enumerable
alice.age = 40;
assert.equal(40, alice.age); // age is writable

// Use defineProperty method to add a property
Object.defineProperty(alice, “salary”, { // still have defineProperties method
value: 1000 // By default, all three properties are false(writable, enumable, configurable);
});

console.log(“**1**“);
console.log(Object.keys(alice)); // value is invisible and thus is not enumerable
assert.equal(1000, alice.salary);
alice.salary = 2000;
assert.equal(1000, alice.salary); // salary is not writable

Object.defineProperty(alice, “address”, {
value: “Princeton”,
enumerable: true
});
console.log(“*2*“);
console.log(Object.keys(alice));
assert.equal(“Princeton”, alice.address); // address is enumerable
alice.address = “Edison”;
assert.equal(“Princeton”, alice.address); // address is not writable

/
Object.defineProperty(alice, “address”, { // Error: address is not configurable because the default is false
value: “Princeton”,
enumerable: true
}); /

Object.defineProperty(alice, “age”, {
writable: false,
enumerable: false
});

console.log(“*3*“);
console.log(Object.keys(alice));
alice.age = 70;
assert.equal(40, alice.age);
