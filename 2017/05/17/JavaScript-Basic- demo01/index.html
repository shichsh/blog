Hoisting Feature

In JavaScript, a variable can be declared after it has been used. It only hoists declarations, not initializations.
```JavaScript
/*
for (int i = 0 ….) { // i ->if i can not access before or after function, it is called block scope

}
*/

(function() {
assert.equal(undefined, i);
assert.equal(undefined, j);
for (var i = 0; i < 10; i++) {
for (var j = 0; j < 5; j++) {
// do something
}
}
assert.equal(10, i);
assert.equal(5, j);
})(); // i and j are hoisted(moved to the top)

/* ES6

  1. let 2.const -> block scope
    */

(function() {
assert.equal(“foo”, foo());
function foo() { // This function definition has hoisting feature because function can be called anywhere
return “foo”;
}
assert.ok(typeof goo == “undefined”);
var goo = function() { // This function definition does not have hoisting feature because this variable is function
return “goo”;
}
assert.equal(“goo”, goo());
});
